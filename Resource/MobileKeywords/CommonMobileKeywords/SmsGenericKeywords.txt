*** Keywords ***
Wait SMS
    [Arguments]    ${os}    ${totalSms}    ${timeout}
    Log    ${os}
    Run Keyword If    '${os}' == 'Android'    SmsMoodKeywords.moodWaitSms    ${totalSms}    ${timeout}
    Run Keyword If    '${os}' == 'iOS'    SmsIosKeywords.iosWaitSms    ${totalSms}    ${timeout}

Open SMS App
    [Arguments]    ${remoteUrl}    ${platformName}    ${platformVersion}    ${udid}
    Log    ${remoteUrl}
    Log    ${platformName}
    Log    ${platformVersion}
    Log    ${udid}
    Run Keyword If    '${platformName}' == 'Android'    mood Open SMS App    ${remoteUrl}    ${platformName}    ${platformVersion}    ${udid}
    Run Keyword If    '${platformName}' == 'iOS'    ios Open SMS App    ${remoteUrl}    ${platformName}    ${platformVersion}    ${udid}
